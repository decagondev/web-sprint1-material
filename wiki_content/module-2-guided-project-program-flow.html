    const a = 1
    const b = null // one of the falsy values (besides undefined, 0, '', NaN)

    // 1- checking truthiness of a single value
    if (a) {
      console.log('a is truthy') // this is the one that runs for this value of a
    } else {
      console.log('a is falsy')
    }
    // 2- checking falsiness of a single value
    if (!b) {
      console.log('b is falsy') // this is what runs for this value of b
    } else {
      console.log('b is truthy')
    }
    // 3- checking truthiness of all values
    if (a && b) {
      console.log('both values are truthy')
    } else {
      console.log('either a or b or both are falsy') // this runs
    }
    // 4 checking truthiness of some values
    if (a || b) {
      console.log('either a or b or both are truthy') // this runs
    } else {
      console.log('both are falsy')
    }
    // 5- checking falsiness of all values
    if (!a && !b) {
      console.log('both falsy')
    } else {
      console.log('one or both are truthy') // this runs
    }
    // 6- checking falsiness of some values
    if (!a || !b) {
      console.log('one or both are falsy') // this runs
    } else {
      console.log('both truthy')
    }
    const points1 = 13
    const points2 = 23
    // 7- check that both values greater than 0 and that points1 bigger than points2
    if (points1 > 0 && points2 > 0 && points1 > points2) {
      console.log('condition met')
    } else {
      console.log('condition not met') // this runs because points points1 < points2
    }
    // 8- check that both values greater than 0 or that points1 bigger than points2
    if ((points1 > 0 && points2 > 0) || points1 > points2) {
      console.log('condition met') // this runs because both values than zero
    } else {
      console.log('condition not met')
    }
    // 9- careful with flawed conditional statements like the following
    const num = 10
    if (num > 0) {
      console.log('bigger than 0')
    } else if (num > 10) {
      console.log('bigger than 10') // never runs
    } else if (num > 20) {
      console.log('bigger than 20') // never runs
    }
    // 10- modulo operator to find if a number is divisible by another
    if (num % 7 === 0) {
      console.log('num divisible by seven (remainder zero)')
    } else {
      console.log(`num not divisible: remainder ${num % 7}`)
    }
    // 11- Logical expressions with OR
    const value1 = 1 || 0 // value is 1, no need to keep going
    const value2 = null || 2 // value is 2, the OR does not give up on the first falsy
    const value3 = null || NaN // value is NaN
    const value4 = null || 6 || NaN // value is 6
    // 12- Logical expressions with AND
    const value5 = 1 && 2 // value is 2, AND continues checking, a falsy would stop it
    const value6 = 1 && null && 67 // value is null, that falsy stopped it
