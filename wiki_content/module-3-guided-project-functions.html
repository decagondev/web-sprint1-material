      }
      return "There is a problem with the code"
    }

      console.log('arg a -->', a)

      console.log('result -->', result)
      return result
    }
    sum(34, 56) // return 34 + 56

    function report(points, level) {
      let status
      if (points > 90) {
        status = "doing great"
      } else if (points > 50) {
        status = "still kickin'"
      } else {
        status = "in bad shape"
      }
      let str = `I'm on level ${level} and I am ${status}!`
      return str
    }
